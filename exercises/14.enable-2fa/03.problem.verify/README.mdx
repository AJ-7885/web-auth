# Verify Code

```tsx
import { generateTOTP } from '@epic-web/totp'

const otpUri = new URL('paste otpUri here')
const { secret, algorithm, digits, period } = Object.fromEntries(
	otpUri.searchParams.entries(),
)

const { otp } = generateTOTP({
	secret,
	algorithm,
	digits,
	period,
})

console.log(otp)
```
